import anthropic
from pathlib import Path
from dotenv import load_dotenv

load_dotenv()
client = anthropic.Anthropic()

# Brand file as system prompt
brand = Path("writing-room/docs/origin-brand.md").read_text()

response = client.messages.create(
  model="claude-opus-4-6",
  max_tokens=2000,
  system=brand,
  messages=[{"role": "user",
             "content": YOUR_PROMPT}]
)
